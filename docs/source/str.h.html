/**
 * @class str_t 
 * 字符串对象。
 */
/**
     * @property {uint32_t} size
     * 字符串的长度。
     */
/**
     * @property {char*} str
     * 字符串的内容。
     */
/**
 * @method str_create
 * 创建str对象。
 *
 * @return {str_t*} str对象。
 */
/**
 * @method str_ref
 * 增加引用计数。
 * @param {str_t*} v str对象。
 *
 * @return {str_t*} str对象本身。
 */
/**
 * @method str_unref
 * 减少引用计数。
 * @param {str_t*} v str对象。
 *
 * @return {str_t*} 引用计数仍然大于0返回str对象本身，否则返回NULL。
 */
