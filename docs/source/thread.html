/**
 * @class thread_t 
 * 线程对象。
 */
/**
 * @method thread_create
 * 创建thread对象。
 *
 * @return {thread_t*} thread对象。
 */
/**
 * @method thread_start
 * 启动线程。
 * @param {thread_t*}    thread thread对象。
 *
 * @return {bool_t} 成功返回TRUE，失败返回FALSE。
 */
/**
 * @method thread_join
 * 等待线程退出。
 * @param {thread_t*}    thread thread对象。
 *
 * @return {bool_t} 成功返回TRUE，失败返回FALSE。
 */
/**
 * @method thread_destroy
 * 销毁thread对象。
 * @param {thread_t*}    thread thread对象。
 *
 * @return {void} 无。
 */
